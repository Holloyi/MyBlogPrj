```
git clone '仓库http地址' //将项目拉到本地
git status //查看状态
git add . //添加全部
git commit -m "description" //commit并添加日志文本
git push orgin main //将本地文件推送至服务器main分支
git pull orgin main //将服务器main分支代码拉到本地
```
